javaScript数值计算系列

June 28, 2017

Reading time ~1 minute

数值计算

  1. 四舍五入保留2位小数

var a = 30;
var b = 26;
//JS保留2位小数：
//toFixed(num),num为几，就四舍五入保留几位小数
var c = (a/b).toFixed(2);
//Math.rund(),四舍五入保留整数

2.对随机数列进行排序 1. 对1–100之间的随机整数进行升序排列 - Math.random()是令系统随机选取大于等于 0.0 且小于 1.0 的随机 double 值


//取1--100之间的10个随机整数进行排序
Math.floor(Math.random()*100+1);//Math.floor取小于这个数的最大整数
var numArr = [];
for(var i = 0; i<10 ; i++){
    numArr[i]=Math.floor(Math.random()*100+1);
}
//sort()可对随机数进行从小到大排序,返回值为排序后的数组
numArr.sort();
  1. 对1–100之间的随机整数进行降序排列

function paixu(m,n){
    return m-n;//m-n为升序，n-m为降序排序
}
var numArr = [];
for(var i = 0; i<10 ; i++){
    numArr[i]=Math.floor(Math.random()*100+1);
}
numArr.sort(paixu);
