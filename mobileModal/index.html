基于jQuery 移动端模态框

July 01, 2017

Reading time ~1 minute

基于jQuery 移动端模态框，防止点击穿透

之前做移动端的项目，一直被模态框的问题困扰了很久，当模态框高度超过浏览器窗口时，滑动模态框，模态框底部html内容也开始滑动，影响用户体验，之后当模态框弹出的时候，给html加了一个height：100%，解决了这个问题。


var modal = {
    open : function(opt,noForbidden){
        var lock = '.lock_overlay';
        $(opt.modal).wrap('<div class="lock_overlay"></div>');  //给模态框外层套上一个div
        $(lock).css({'display':'block'});
        $(opt.modal).css({'display':'block'});
        $(opt.modal).fadeIn();  
        $('html,body').addClass('modal');  //给html和body加上modal，单独给modal写样式
    },
    close : function(opt){
        var lock = '.lock_overlay';
        $(lock).css({'display':'none'});
        $(opt.modal).unwrap('<div class="lock_overlay"></div>');
        $(opt.modal).css({'display':'none'});
        $('html,body').removeClass('modal');
    }
} 


html {
    width: 750px;
}

body {
    margin: 0;
    background: lightblue;
}

.modal {
    height: 100%;
    overflow: hidden;
}

.hide {
    display: none;
}

.lock_overlay {
    display: none;
    position: fixed;
    top: 0;
    left: 0;
    bottom: 0;
    right: 0;
    z-index: 1001;
    background: rgba(0, 0, 0, 0.2);
    overflow-y: auto;
    overflow-x: hidden;
}
